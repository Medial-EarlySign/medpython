117inline uint32_t hash_space(vw& all, std::string s)
118{ substring ss;
119 ss.begin = (char*)s.c_str();
120 ss.end = ss.begin + s.length();
121 return (uint32_t)all.p->hasher(ss,hash_base);
122}
123inline uint32_t hash_space_static(std::string s, std::string hash)
124{ substring ss;
125 ss.begin = (char*)s.c_str();
126 ss.end = ss.begin + s.length();
127 return (uint32_t)getHasher(hash)(ss, hash_base);
128}
129//Then use it as the seed for hashing features.
130inline uint32_t hash_feature(vw& all, std::string s, uint64_t u)
131{ substring ss;
132 ss.begin = (char*)s.c_str();
133 ss.end = ss.begin + s.length();
134 return (uint32_t)(all.p->hasher(ss,u) & all.parse_mask);
135}
136inline uint32_t hash_feature_static(std::string s, unsigned long u, std::string h, uint32_t num_bits)
137{ substring ss;
138 ss.begin = (char*)s.c_str();
139 ss.end = ss.begin + s.length();
140 size_t parse_mark = (1 << num_bits) - 1;
141 return (uint32_t)(getHasher(h)(ss, u) & parse_mark);
142}

144inline uint32_t hash_feature_cstr(vw& all, char* fstr, unsigned long u)
145{ substring ss;
146 ss.begin = fstr;
147 ss.end = ss.begin + strlen(fstr);
148 return (uint32_t)(all.p->hasher(ss,u) & all.parse_mask);
