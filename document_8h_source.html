854 template <typename SourceAllocator>
855 bool operator==(const GenericValue<Encoding, SourceAllocator>& rhs) const
856 { typedef GenericValue<Encoding, SourceAllocator> RhsType;
857 if (GetType() != rhs.GetType())
858 return false;
859
860 switch (GetType())
861 { case kObjectType: // Warning: O(n^2) inner-loop
862 if (data_.o.size != rhs.data_.o.size)
863 return false;
864 for (ConstMemberIterator lhsMemberItr = MemberBegin(); lhsMemberItr != MemberEnd(); ++lhsMemberItr)
865 { typename RhsType::ConstMemberIterator rhsMemberItr = rhs.FindMember(lhsMemberItr->name);
866 if (rhsMemberItr == rhs.MemberEnd() || lhsMemberItr->value != rhsMemberItr->value)
867 return false;
868 }
869 return true;
870
871 case kArrayType:
872 if (data_.a.size != rhs.data_.a.size)
873 return false;
874 for (SizeType i = 0; i < data_.a.size; i++)
875 if ((*this)[i] != rhs[i])
876 return false;
877 return true;
878
879 case kStringType:
880 return StringEqual(rhs);
881
882 case kNumberType:
883 if (IsDouble() || rhs.IsDouble())
884 { double a = GetDouble(); // May convert from integer to double.
885 double b = rhs.GetDouble(); // Ditto
886 return a >= b && a <= b; // Prevent -Wfloat-equal
887 }
888 else
889 return data_.n.u64 == rhs.data_.n.u64;
890
891 default:
892 return true;
893 }

954 // Checks whether a number can be losslessly converted to a double.
955 bool IsLosslessDouble() const
956 { if (!IsNumber()) return false;
957 if (IsUint64())
958 { uint64_t u = GetUint64();
959 volatile double d = static_cast<double>(u);
960 return (d >= 0.0)
961 && (d < static_cast<double>(std::numeric_limits<uint64_t>::max()))
962 && (u == static_cast<uint64_t>(d));
963 }
964 if (IsInt64())
965 { int64_t i = GetInt64();
966 volatile double d = static_cast<double>(i);
967 return (d >= static_cast<double>(std::numeric_limits<int64_t>::min()))
968 && (d < static_cast<double>(std::numeric_limits<int64_t>::max()))
969 && (i == static_cast<int64_t>(d));
970 }
971 return true; // double, int, uint are always lossless
972 }

985 if (a < static_cast<double>(-std::numeric_limits<float>::max())
986 || a > static_cast<double>(std::numeric_limits<float>::max()))
987 return false;

1770 template <typename Handler>
1771 bool Accept(Handler& handler) const
1772 { switch(GetType())
1773 { case kNullType: return handler.Null();
1774 case kFalseType: return handler.Bool(false);
1775 case kTrueType: return handler.Bool(true);
1776
1777 case kObjectType:
1778 if (RAPIDJSON_UNLIKELY(!handler.StartObject()))
1779 return false;
1780 for (ConstMemberIterator m = MemberBegin(); m != MemberEnd(); ++m)
1781 { RAPIDJSON_ASSERT(m->name.IsString()); // User may change the type of name by MemberIterator.
1782 if (RAPIDJSON_UNLIKELY(!handler.Key(m->name.GetString(), m->name.GetStringLength(), (m->name.data_.f.flags & kCopyFlag) != 0)))
1783 return false;
1784 if (RAPIDJSON_UNLIKELY(!m->value.Accept(handler)))
1785 return false;
1786 }
1787 return handler.EndObject(data_.o.size);
1788
1789 case kArrayType:
1790 if (RAPIDJSON_UNLIKELY(!handler.StartArray()))
1791 return false;
1792 for (const GenericValue* v = Begin(); v != End(); ++v)
1793 if (RAPIDJSON_UNLIKELY(!v->Accept(handler)))
1794 return false;
1795 return handler.EndArray(data_.a.size);
1796
1797 case kStringType:
1798 return handler.String(GetString(), GetStringLength(), (data_.f.flags & kCopyFlag) != 0);
1799
1800 default:
1801 RAPIDJSON_ASSERT(GetType() == kNumberType);
1802 if (IsDouble()) return handler.Double(data_.n.d);
1803 else if (IsInt()) return handler.Int(data_.n.i.i);
1804 else if (IsUint()) return handler.Uint(data_.n.u.u);
1805 else if (IsInt64()) return handler.Int64(data_.n.i64);
1806 else return handler.Uint64(data_.n.u64);
1807 }
