159 //k = 2;

163
164 boost::math::chi_squared_distribution<> chi_dist(nu);
165 double ten_pctile_chisq_df_hyper_nu = boost::math::cdf(chi_dist, 1 - 0.9);
166 lambda = sample_var_y_in_data * ten_pctile_chisq_df_hyper_nu / nu;
167 //lambda = 2.7e-3;
