134class namedlabels
135{
136private:
137
138 v_array<substring> id2name;
139 v_hashmap<substring,uint64_t> name2id;
140 uint32_t K;
141
142public:
143
144 namedlabels(std::string label_list)
145 { id2name = v_init<substring>();
146 char* temp = calloc_or_throw<char>(1+label_list.length());
147 memcpy(temp, label_list.c_str(), strlen(label_list.c_str()));
148 substring ss = { temp, nullptr };
149 ss.end = ss.begin + label_list.length();
150 tokenize(',', ss, id2name);
151
152 K = (uint32_t)id2name.size();
153 name2id.delete_v();//delete automatically allocated vector.
154 name2id.init(4 * K + 1, 0, substring_equal);
155 for (size_t k=0; k<K; k++)
156 { substring& l = id2name[k];
157 uint64_t hash = uniform_hash((unsigned char*)l.begin, l.end-l.begin, 378401);
158 uint64_t id = name2id.get(l, hash);
159 if (id != 0) // TODO: memory leak: char* temp
160 THROW("error: label dictionary initialized with multiple occurances of: " << l);
161 size_t len = l.end - l.begin;
162 substring l_copy = { calloc_or_throw<char>(len), nullptr };
163 memcpy(l_copy.begin, l.begin, len * sizeof(char));
164 l_copy.end = l_copy.begin + len;
165 name2id.put(l_copy, hash, (uint32_t)(k+1));
166 }
167 }
168
169 ~namedlabels()
170 { if (id2name.size()>0)
171 free(id2name[0].begin);
172 name2id.iter(deleter);
173 name2id.delete_v();
174 id2name.delete_v();
175 }
176
177 uint32_t getK() { return K; }
178
179 uint64_t get(substring& s)
180 { uint64_t hash = uniform_hash((unsigned char*)s.begin, s.end-s.begin, 378401);
181 uint64_t v = name2id.get(s, hash);
182 if (v == 0)
183 { std::cerr << "warning: missing named label '";
184 for (char*c = s.begin; c != s.end; c++) std::cerr << *c;
185 std::cerr << '\'' << std::endl;
186 }
187 return v;
188 }
189
190 substring get(uint32_t v)
191 { if ((v == 0) || (v > K))
192 { substring ss = {nullptr,nullptr};
193 return ss;
194 }
195 else
196 return id2name[v-1];
197 }
198};

258 { t += weight;
259 if(test_example)
260 { weighted_holdout_examples += weight;//test weight seen
261 weighted_holdout_examples_since_last_dump += weight;
262 weighted_holdout_examples_since_last_pass += weight;

270 weighted_labeled_examples += weight;
271 else
272 weighted_unlabeled_examples += weight;
273 sum_loss += loss;
274 sum_loss_since_last_dump += loss;
275 total_features += num_features;
