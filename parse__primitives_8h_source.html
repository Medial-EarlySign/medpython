59// The following function is a home made strtof. The
60// differences are :
61// - much faster (around 50% but depends on the string to parse)
62// - less error control, but utilised inside a very strict parser
63// in charge of error detection.
64inline float parseFloat(char * p, char **end)
65{ char* start = p;
66
67 if (!*p)
68 { *end = p;
69 return 0;
70 }
71 int s = 1;
72 while (*p == ' ') p++;
73
74 if (*p == '-')
75 { s = -1; p++;
76 }
77
78 float acc = 0;
79 while (*p >= '0' && *p <= '9')
80 acc = acc * 10 + *p++ - '0';
81
82 int num_dec = 0;
83 if (*p == '.')
84 { while (*(++p) >= '0' && *p <= '9')
85 { if (num_dec < 35)
86 { acc = acc *10 + (*p - '0');
87 num_dec++;
88 }
89 }
90 }
91
92 int exp_acc = 0;
93 if(*p == 'e' || *p == 'E')
94 { p++;
95 int exp_s = 1;
96 if (*p == '-')
97 { exp_s = -1; p++;
98 }
99 while (*p >= '0' && *p <= '9')
100 exp_acc = exp_acc * 10 + *p++ - '0';
101 exp_acc *= exp_s;
102
103 }
104 if (*p == ' ' || *p == '\n' || *p == '\t')//easy case succeeded.
105 { acc *= powf(10,(float)(exp_acc-num_dec));
106 *end = p;
107 return s * acc;
108 }
109 else
110 return (float)strtod(start,end);
111}

116inline float float_of_substring(substring s)
117{ char* endptr = s.end;
118 float f = parseFloat(s.begin,&endptr);
119 if ((endptr == s.begin && s.begin != s.end) || nanpattern(f))
120 { std::cout << "warning: " << std::string(s.begin, s.end-s.begin).c_str() << " is not a good float, replacing with 0" << std::endl;
121 f = 0;
122 }
123 return f;
124}
