217 inline bool operator< (const PredictedRes &pr2) const {
218 if (split == pr2.split) {
219 if (pid == pr2.pid) return (date < pr2.date);
220 return (pid < pr2.pid);
221 }
222 return (split < pr2.split);
223 };
