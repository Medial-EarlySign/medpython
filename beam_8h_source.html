30inline int compare_on_cost(const void *void_a, const void *void_b)
31{ if (void_a == void_b) return 0;
32 const beam_element<void> *a = (const beam_element<void>*) void_a;
33 const beam_element<void> *b = (const beam_element<void>*) void_b;
34 if ( a->active && !b->active) return -1; // active things come before inactive things
35 else if (!a->active && b->active) return 1;
36 else if (!a->active && !b->active) return 0;
37 else if (a->cost < b->cost) return -1; // otherwise sort by cost
38 else if (a->cost > b->cost) return 1;
39 else return 0;
40}
41
42inline int compare_on_hash_then_cost(const void *void_a, const void *void_b)
43{ if (void_a == void_b) return 0;
44 const beam_element<void> *a = (const beam_element<void>*) void_a;
45 const beam_element<void> *b = (const beam_element<void>*) void_b;
46 if ( a->active && !b->active) return -1; // active things come before inactive things
47 else if (!a->active && b->active) return 1;
48 else if (!a->active && !b->active) return 0;
49 else if (a->hash < b->hash) return -1; // if the hashes are different, sort by hash
50 else if (a->hash > b->hash) return 1;
51 else if (a->cost < b->cost) return -1; // otherwise sort by cost
52 else if (a->cost > b->cost) return 1;
53 else return 0;
54}
