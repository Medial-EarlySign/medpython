48template <bool is_learn>
49void gen_cs_example_dm(cb_to_cs& c, example& ec, COST_SENSITIVE::label& cs_ld)
50{ //this implements the direct estimation method, where costs are directly specified by the learned regressor.
51 CB::label ld = ec.l.cb;
52
53 float min = FLT_MAX;
54 uint32_t argmin = 1;
55 //generate cost sensitive example
56 cs_ld.costs.erase();
57 c.pred_scores.costs.erase();
58
59 if (ld.costs.size() == 1 && !is_test_label(ld)) //this is a typical example where we can perform all actions
60 { //in this case generate cost-sensitive example with all actions
61 for (uint32_t i = 1; i <= c.num_actions; i++)
62 { COST_SENSITIVE::wclass wc = {0., i, 0., 0.};
63 //get cost prediction for this action
64 wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, c.known_cost, ec, i, 0);
65 if (wc.x < min)
66 { min = wc.x;
67 argmin = i;
68 }
69
70 c.pred_scores.costs.push_back(wc);
71
72 if (c.known_cost != nullptr && c.known_cost->action == i)
73 { c.nb_ex_regressors++;
74 c.avg_loss_regressors += (1.0f / c.nb_ex_regressors)*((c.known_cost->cost - wc.x)*(c.known_cost->cost - wc.x) - c.avg_loss_regressors);
75 c.last_pred_reg = wc.x;
76 c.last_correct_cost = c.known_cost->cost;
77 }
78
79 cs_ld.costs.push_back(wc);
80 }
81 }
82 else //this is an example where we can only perform a subset of the actions
83 { //in this case generate cost-sensitive example with only allowed actions
84 for (auto& cl : ld.costs)
85 { COST_SENSITIVE::wclass wc = {0., cl.action, 0., 0.};
86
87 //get cost prediction for this action
88 wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, c.known_cost, ec, cl.action, 0);
89 if (wc.x < min || (wc.x == min && cl.action < argmin))
90 { min = wc.x;
91 argmin = cl.action;
92 }
93 c.pred_scores.costs.push_back(wc);
94
95 if (c.known_cost != nullptr && c.known_cost->action == cl.action)
96 { c.nb_ex_regressors++;
97 c.avg_loss_regressors += (1.0f / c.nb_ex_regressors)*((c.known_cost->cost - wc.x)*(c.known_cost->cost - wc.x) - c.avg_loss_regressors);
98 c.last_pred_reg = wc.x;
99 c.last_correct_cost = c.known_cost->cost;
100 }
101
102 cs_ld.costs.push_back(wc);
103 }
104 }
105
106 ec.pred.multiclass = argmin;
107}

109template <bool is_learn>
110void gen_cs_label(cb_to_cs& c, example& ec, COST_SENSITIVE::label& cs_ld, uint32_t action)
111{ COST_SENSITIVE::wclass wc = {0., action, 0., 0.};
112
113 //get cost prediction for this action
114 wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, c.known_cost, ec, action, c.num_actions);
115
116 c.pred_scores.costs.push_back(wc);
117 //add correction if we observed cost for this action and regressor is wrong
118 if (c.known_cost != nullptr && c.known_cost->action == action)
119 { c.nb_ex_regressors++;
120 c.avg_loss_regressors += (1.0f / c.nb_ex_regressors)*((c.known_cost->cost - wc.x)*(c.known_cost->cost - wc.x) - c.avg_loss_regressors);
121 c.last_pred_reg = wc.x;
122 c.last_correct_cost = c.known_cost->cost;
123 wc.x += (c.known_cost->cost - wc.x) / c.known_cost->probability;
124 }
125
126 cs_ld.costs.push_back(wc);

130template <bool is_learn>
131void gen_cs_example_dr(cb_to_cs& c, example& ec, CB::label& ld, COST_SENSITIVE::label& cs_ld)
132{ //this implements the doubly robust method
133 cs_ld.costs.erase();
134 c.pred_scores.costs.erase();
135 if (ld.costs.size() == 0)//a test example
136 for (uint32_t i = 1; i <= c.num_actions; i++)
137 { //Explicit declaration for a weak compiler.
138 COST_SENSITIVE::wclass c = { FLT_MAX, i, 0., 0. };
139 cs_ld.costs.push_back(c);
140 }
141 else if (ld.costs.size() == 1 && !is_test_label(ld)) //this is a typical example where we can perform all actions
142 //in this case generate cost-sensitive example with all actions
143 for (uint32_t i = 1; i <= c.num_actions; i++)
144 { gen_cs_label<is_learn>(c, ec, cs_ld, i);
145 }
146 else //this is an example where we can only perform a subset of the actions
147 //in this case generate cost-sensitive example with only allowed actions
148 for (auto& cl : ld.costs)
149 gen_cs_label<is_learn>(c, ec, cs_ld, cl.action);
150}

177template <bool is_learn>
178void gen_cs_example_dr(cb_to_cs_adf& c, v_array<example*> examples, COST_SENSITIVE::label& cs_labels)
179{ //size_t mysize = examples.size();
180 c.pred_scores.costs.erase();
181 bool shared = CB::ec_is_example_header(*examples[0]);
182 int startK = 0;
183 if (shared) startK = 1;
184
185 cs_labels.costs.erase();
186 for (size_t i = 0; i < examples.size(); i++)
187 { if (CB_ALGS::example_is_newline_not_header(*examples[i])) continue;
188
189 COST_SENSITIVE::wclass wc = {0.,0,0.,0.};
190
191 if (c.known_cost.action + startK == i)
192 { int known_index = c.known_cost.action;
193 c.known_cost.action = 0;
194 //get cost prediction for this label
195 // num_actions should be 1 effectively.
196 // my get_cost_pred function will use 1 for 'index-1+base'
197 wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, &(c.known_cost), *(examples[i]), 0, 2);
198 c.known_cost.action = known_index;
199 }
200 else
201 wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, nullptr, *(examples[i]), 0, 2);
202
203 if (shared)
204 wc.class_index = (uint32_t)i - 1;
205 else
206 wc.class_index = (uint32_t)i;
207 c.pred_scores.costs.push_back(wc); // done
208
209 //add correction if we observed cost for this action and regressor is wrong
210 if (c.known_cost.probability != -1 && c.known_cost.action + startK == i)
211 wc.x += (c.known_cost.cost - wc.x) / c.known_cost.probability;
212 cs_labels.costs.push_back(wc);
213 }
214
215 if (shared)//take care of shared examples
216 { cs_labels.costs[0].class_index = 0;
217 cs_labels.costs[0].x = -FLT_MAX;
218 }
219}

238template<bool is_learn>
239void call_cs_ldf(LEARNER::base_learner& base, v_array<example*>& examples, v_array<CB::label>& cb_labels,
240 COST_SENSITIVE::label& cs_labels, v_array<COST_SENSITIVE::label>& prepped_cs_labels, uint64_t offset, size_t id = 0)
241{ cb_labels.erase();
242 if (prepped_cs_labels.size() < cs_labels.costs.size()+1)
243 { prepped_cs_labels.resize(cs_labels.costs.size()+1);
244 prepped_cs_labels.end() = prepped_cs_labels.end_array;
245 }
246
247 // 1st: save cb_label (into mydata) and store cs_label for each example, which will be passed into base.learn.
248 size_t index = 0;
249 for (example* ec : examples)
250 { cb_labels.push_back(ec->l.cb);
251 prepped_cs_labels[index].costs.erase();
252 if (index != examples.size()-1)
253 prepped_cs_labels[index].costs.push_back(cs_labels.costs[index]);
254 else
255 prepped_cs_labels[index].costs.push_back({FLT_MAX,0,0.,0.});
256 ec->l.cs = prepped_cs_labels[index++];
257 }
258
259 // 2nd: predict for each ex
260 // // call base.predict for each vw exmaple in the sequence
261 for (example* ec : examples)
262 { uint64_t old_offset = ec->ft_offset;
263 ec->ft_offset = offset;
264 if (is_learn)
265 base.learn(*ec, id);
266 else
267 base.predict(*ec, id);
268 ec->ft_offset = old_offset;
269
270 }
271 // 3rd: restore cb_label for each example
272 // (**ec).l.cb = array.element.
273 size_t i = 0;
274 for (example* ec : examples)
275 ec->l.cb = cb_labels[i++];
276}
