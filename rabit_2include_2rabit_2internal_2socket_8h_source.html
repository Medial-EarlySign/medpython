168 int ret = PollImpl(fdset.data(), fdset.size(), timeout);
169 if (ret == 0) {
170 LOG(FATAL) << "Poll timeout";
171 } else if (ret < 0) {
172 LOG(FATAL) << "Failed to poll.";
173 } else {

175 auto revents = pfd.revents & pfd.events;
176 if (!revents) {
177 fds.erase(pfd.fd);
