32{
33 ifstream inf;
34 unsigned long long size;
35
36 data.clear();
37 size = get_file_size(fname);
38
39 if (start_pos >= size)
40 return 0;
41
42 inf.open(fname, ios::in | ios::binary);
43 if (!inf)
44 return -1;
45
46 inf.seekg(start_pos, ios_base::beg);
47
48 size = size - start_pos;
49 if (size > 0) {
50 data.resize(size / (sizeof(T)));
51
52 char *d = (char *)&data[0];
53 inf.read(d, size);
54 }
55 return 0;
56}

65//a class for reading separated files (csv, tsv, etc.) with mixed data types
66//the tuple of specified types should be provided as a template type. Integral types are checked for correctness/missingness
67//and replaced with special values accordingly: incorrect as the type's min(), missing as the type's max(). all fields are trimmed for white space.
68//int the untyped case, all the fields are read as strings
69//in both cases, separators inside quotes are ignored. quotes inside quotes are expected to be doubled.
70
71//Example use:
72
73//with type specification:
74/*
75using Type = tuple<int, int, int, string, string, int, int, string, string, int, string, string, string, string>;
76string fname = "//nas1/Data/eicu/pastHistory.csv";
77int monitorStep = 100000;
78string fname = //nas1/Data/eicu/pastHistory.csv;
79SepFileReader<Type> sr(fname, ',', 0, true, monitorStep);
80sr.read();
81const vector<Type>& data = sr.getData();
82*/
83
84//without type specification:
85/*
86SepFileReader<string> sr1(fname, ',', 0, true, monitorStep);
87sr1.read();
88vector<vector<string> > data1 = sr1.getData();
89*/

95//helper class for _read
96//read a single position in a single record string into an element of a tuple, given a pair of positions
97//where a field starts and ends
98template <typename Tuple, size_t N>
99struct handleSinglePos {
100 void operator()(const string& str, const pair<int, int>& pos1, Tuple& e) {
101
102 pair<int, int> pos = pos1;
103
104 //trim whitespace
105 const char *buf = str.c_str();
106 while (pos.first <= pos.second && isspace(buf[pos.first]))
107 pos.first++;
108
109 while (pos.first <= pos.second && isspace(buf[pos.second]))
110 pos.second--;
111
112 //handle missing value
113 if (pos.first > pos.second)
114 SepFileReader<Tuple>::setMissingVal(get<N>(e));
115 else {
116 stringstream s(str.substr(pos.first, pos.second - pos.first + 1));
117 s >> get<N>(e);
118
119 if (s.fail())
120 SepFileReader<Tuple>::setBadVal(get<N>(e));
121 }
122 };
123};

125//helper class for _read
126//read a single record string into a tuple, given pairs of positions
127//where fields start and end
128template <typename Tuple, size_t N>
129struct handleSepRecord {
130 void operator()(const string& str, const vector<pair<int, int> >& pos, Tuple& e) {
131 handleSinglePos<Tuple, N>()(str, pos[N], e);
132 handleSepRecord<Tuple, N - 1>()(str, pos, e);
133 };
134};

136//specializetion of handleSepRecord for N=0
137template <typename Tuple>
138struct handleSepRecord<Tuple, 0> {
139 void operator()(const string& str, const vector<pair<int, int> >& pos, Tuple& e) {
140 handleSinglePos<Tuple, 0>()(str, pos[0], e);
141 };

144//------------- the main class -------------
145template <typename Tuple>
146class SepFileReader {
147public:
148 //constructor:
149 //sep - separator character
150 //maxRecords - max records to read. default 0 (all).
151 //skipHeader - whether or not to skip a single header line
152 //monitorStep - each <monitorStep> records, print out a '.' to monitor progress
153 SepFileReader(const string& fname1, char sep1 = ',', size_t maxRecords1 = 0, bool skipHeader1 = true, int monitorStep1 = 0) :
154 fname(fname1), sep(sep1), maxRecords(maxRecords1), skipHeader(skipHeader1), monitorStep(monitorStep1) {}
155
156 //read the file
157 //returns - 0 on success, -1 on failure.
158 int read() { return _read<tuple_size<Tuple>::value>(); }
159 vector<Tuple>& getData() { return v; }
160 //set variable to its type's missing value
161 //for integral types it's the max(), otherwise nothing is done
162 template <typename T>
163 static void setMissingVal(T& val) {
164 if (is_arithmetic<T>::value)
165 val = numeric_limits<T>::max();
166 }
167
168 //for arithmetic types, set val to the min of the type, signifying bad val
169 //for other types, do nothing
170 template <typename T>
171 static void setBadVal(T& val) {
172 if (is_arithmetic<T>::value)
173 val = numeric_limits<T>::min();
174 }
175private:
176 vector<Tuple> v;
177 string fname;
178 char sep;
179 size_t maxRecords;
180 bool skipHeader;
181 int monitorStep;
182
183 //does main work of read()
184 template<size_t N>
185 int _read() {
186 v.clear();
187 const size_t DEFAULT_NUM_RECORDS = 1000000;
188
189 //reserve proper space for v
190 if (maxRecords == 0) {
191 v.reserve(DEFAULT_NUM_RECORDS);
192 maxRecords = numeric_limits<size_t>::max();
193 }
194 else
195 v.reserve(maxRecords);
196
197 //open file
198 ifstream inf(fname);
199 if (!inf.is_open()) {
200 cerr << "Cannot open " << fname << " for reading" << endl;
201 return -1;
202 }
203
204 std::string str;
205 Tuple elem;
206 vector<pair<int, int> > positions; //hold the positions of fields (begin and end)
207
208 //handle header line correctly
209 if (skipHeader && !inf.eof())
210 getline(inf, str);
211
212 //go over lines
213 while (!inf.eof() && v.size() < maxRecords) {
214
215 //monitor progress
216 if (monitorStep > 0 && v.size() % monitorStep == 0)
217 cout << ".";
218
219 //process a single line
220 getline(inf, str);
221 if (str.empty())
222 break;
223
224 const char *buf = str.c_str();
225 positions.clear();
226
227 int curStart = 0;
228 bool inQuotes = false;
229
230 for (int i = 0; i < str.length(); ++i) {
231 if (buf[i] == '"') {
232 inQuotes = !inQuotes;
233 continue;
234 }
235
236 if (!inQuotes && buf[i] == sep) {
237 positions.push_back(pair<int, int>(curStart, i - 1));
238 curStart = i + 1;
239 }
240 }
241
242 positions.push_back(pair<int, int>(curStart, (int)(str.length()) - 1));
243 assert(positions.size() == N);
244
245 handleSepRecord<Tuple, N - 1>()(str, positions, elem);
246 v.push_back(elem);
247 }
248
249 if (monitorStep > 0)
250 cout << endl;
251
252 inf.close();
253 v.shrink_to_fit();
254
255 return 0;
256 }

259//a specialization of SepFileReader that reads separated files
260//treating all fields read as strings. The data is held as
261//a vector<vector<string> > rather than a vector of a specified tuple
262//this setting is preferable if the composition and number of the fields
263//is not important. no missing/correctness checks are performed
264//
265//note: this specialization is quicker, and more memory-expensive.
266template<>
267class SepFileReader<string> {
268public:
269 //constructor
270 SepFileReader(const string& fname1, char sep1 = ',', size_t maxRecords1 = 0, bool skipHeader1 = true, int monitorStep1 = 0) :
271 fname(fname1), sep(sep1), maxRecords(maxRecords1), skipHeader(skipHeader1), monitorStep(monitorStep1) {}
272
273 //count the number of fields in the file, using the first line
274 //if file does not open return -1. if zero lines, return -2.
275 int getNumberFieldsDirectly() const {
276 ifstream inf(fname);
277 if (!inf.is_open())
278 return -1;
279
280 if (inf.eof())
281 return -2;
282
283 string str;
284 getline(inf, str);
285 inf.close();
286
287 if (str.empty())
288 return 0;
289
290 const char *buf = str.c_str();
291 int count = 0;
292 bool inQuotes = false;
293
294 for (int i = 0; i < str.length(); ++i) {
295 if (buf[i] == '"') {
296 inQuotes = !inQuotes;
297 continue;
298 }
299
300 if (!inQuotes && buf[i] == sep)
301 count++;
302 }
303
304 return count + 1;
305 }

307 //return const reference to the data read
308 vector<vector<string> > & getData() { return v; }
309
310 //read the data
311 int read() {
312 int numFields = getNumberFieldsDirectly();
313
314 if (numFields < 1)
315 return -1;
316
317 v.clear();
318 const size_t DEFAULT_NUM_RECORDS = 1000000;
319
320 if (maxRecords == 0) {
321 v.reserve(DEFAULT_NUM_RECORDS);
322 maxRecords = numeric_limits<size_t>::max();
323 }
324 else
325 v.reserve(maxRecords);
326
327 ifstream inf(fname);
328 if (!inf.is_open()) {
329 cerr << "Cannot open " << fname << " for reading" << endl;
330 return -1;
331 }
332
333 string str;
334 vector<string> record; //a single record
335 record.reserve(numFields);
336
337 //handle header line correctly
338 if (skipHeader && !inf.eof())
339 getline(inf, str);
340
341 //go over records
342 while (!inf.eof() && v.size() < maxRecords) {
343 if (monitorStep > 0 && v.size() % monitorStep == 0)
344 cout << ".";
345 //process a single line
346 getline(inf, str);
347 if (str.empty())
348 break;
349
350 const char *buf = str.c_str();
351 record.clear();
352
353 int curStart = 0;
354 bool inQuotes = false;
355
356 for (int i = 0; i < str.length(); ++i) {
357 if (buf[i] == '"') {
358 inQuotes = !inQuotes;
359 continue;
360 }
361
362 if (!inQuotes && buf[i] == sep) {
363 record.push_back(str.substr(curStart, i - curStart));
364 curStart = i + 1;
365 }
366 }
367
368 record.push_back(str.substr(curStart, str.length() - curStart));
369
370 //trim white space
371 for (auto& s : record) {
372 int i = 0, j = (int)(s.length()) - 1;
373 const char *buf = s.c_str();
374 while (i <= j && isspace(buf[i])) i++;
375 while (i <= j && isspace(buf[j])) j--;
376 s = s.substr(i, j - i + 1);
377 s.shrink_to_fit();
378 }
379
380 assert(record.size() == numFields);
381 v.push_back(record);
382 }
383
384 if (monitorStep > 0)
385 cout << endl;
386
387 inf.close();
388 v.shrink_to_fit();
389
390 return 0;
391 }
