74// state data used in non-recursive feature generation algorithm
75// contains N feature_gen_data records (where N is length of interaction)
76struct feature_gen_data
77{ size_t loop_idx; // current feature id in namespace
78 uint64_t hash; // hash of feature interactions of previous namespaces in the list
79 float x; // value of feature interactions of previous namespaces in the list
80 size_t loop_end; // last feature id. May be less than number of features if namespace involved in interaction more than once
81 // calculated at preprocessing together with same_ns
82 size_t self_interaction; // namespace interacting with itself
83 features* ft_arr;
84// feature_gen_data(): loop_idx(0), x(1.), loop_end(0), self_interaction(false) {}
85};

87// The inline function below may be adjusted to change the way
88// synthetic (interaction) features' values are calculated, e.g.,
89// fabs(value1-value2) or even value1>value2?1.0:-1.0
90// Beware - its result must be non-zero.
91inline float INTERACTION_VALUE(float value1, float value2) { return value1*value2; }

98template <class R, class S, void(*T)(R&, float, S), bool audit, void(*audit_func)(R&, const audit_strings*), class W>
99inline void inner_kernel(R& dat, features::iterator_all& begin, features::iterator_all& end, const uint64_t offset, W& weights, feature_value ft_value, feature_index halfhash)
100{
101 if (audit)
102 {
103 for (; begin != end; ++begin)
104 {
105 audit_func(dat, begin.audit().get());
106 call_T<R, T>(dat, weights, INTERACTION_VALUE(ft_value, begin.value()), (begin.index() ^ halfhash) + offset);
107 audit_func(dat, nullptr);
108 }
109 }
110 else
111 { for (; begin != end; ++begin)
112 call_T<R, T>(dat, weights, INTERACTION_VALUE(ft_value, begin.value()), (begin.index() ^ halfhash) + offset);
113 }
114}

121 inline void generate_interactions(vw& all, example& ec, R& dat, W& weights) // default value removed to eliminate ambiguity in old complers
122 {
123 features* features_data = ec.feature_space;
124
125 // often used values
126 const uint64_t offset = ec.ft_offset;
127// const uint64_t stride_shift = all.stride_shift; // it seems we don't need stride shift in FTRL-like hash
128
129 // statedata for generic non-recursive iteration
130 v_array<feature_gen_data > state_data = v_init<feature_gen_data >();
131
132 feature_gen_data empty_ns_data; // micro-optimization. don't want to call its constructor each time in loop.
133 empty_ns_data.loop_idx = 0;
134 empty_ns_data.x = 1.;
135 empty_ns_data.loop_end = 0;
136 empty_ns_data.self_interaction = false;
137
138 // loop throw the set of possible interactions
139 for (v_string& ns : all.interactions)
140 { // current list of namespaces to interact.
141
142#ifndef GEN_INTER_LOOP
143
144 // unless GEN_INTER_LOOP is defined we use nested 'for' loops for interactions length 2 (pairs) and 3 (triples)
145 // and generic non-recursive algorythm for all other cases.
146 // nested 'for' loops approach is faster, but can't be used for interation of any length.
147
148 const size_t len = ns.size();
149
150 if (len == 2) //special case of pairs
151 {
152 features& first = features_data[ns[0]];
153 if (first.nonempty())
154 {
155 features& second = features_data[ns[1]];
156 if (second.nonempty())
157 {
158 const bool same_namespace = ( !all.permutations && ( ns[0] == ns[1] ) );
159
160 for(size_t i = 0; i < first.indicies.size(); ++i)
161 { feature_index halfhash = FNV_prime * (uint64_t)first.indicies[i];
162 if (audit) audit_func(dat, first.space_names[i].get());
163 // next index differs for permutations and simple combinations
164 feature_value ft_value = first.values[i];
165 features::features_value_index_audit_range range = second.values_indices_audit();
166 features::iterator_all begin = range.begin();
167 if (same_namespace)
168 begin += (PROCESS_SELF_INTERACTIONS(ft_value)) ? i : i + 1;
169
170 features::iterator_all end = range.end();
171 inner_kernel<R, S, T, audit, audit_func>(dat, begin, end, offset, weights, ft_value, halfhash);
172
173 if (audit) audit_func(dat, nullptr);
174 } // end for(fst)
175 } // end if (data[snd] size > 0)
176 } // end if (data[fst] size > 0)
177 }
178 else if (len == 3) // special case for triples
179 { features& first = features_data[ns[0]];
180 if (first.nonempty())
181 { features& second = features_data[ns[1]];
182 if (second.nonempty())
183 { features& third = features_data[ns[2]];
184 if (third.nonempty())
185 { // don't compare 1 and 3 as interaction is sorted
186 const bool same_namespace1 = ( !all.permutations && ( ns[0] == ns[1] ) );
187 const bool same_namespace2 = ( !all.permutations && ( ns[1] == ns[2] ) );
188
189 for(size_t i = 0; i < first.indicies.size(); ++i)
190 { if(audit) audit_func(dat, first.space_names[i].get());
191 const uint64_t halfhash1 = FNV_prime * (uint64_t)first.indicies[i];
192 const float& first_ft_value = first.values[i];
193 size_t j=0;
194 if (same_namespace1) // next index differs for permutations and simple combinations
195 j = (PROCESS_SELF_INTERACTIONS(first_ft_value)) ? i : i+1;
196
197 for (; j < second.indicies.size(); ++j)
198 { //f3 x k*(f2 x k*f1)
199 if(audit) audit_func(dat, second.space_names[j].get());
200 feature_index halfhash = FNV_prime * (halfhash1 ^ (uint64_t)second.indicies[j]);
201 feature_value ft_value = INTERACTION_VALUE(first_ft_value, second.values[j]);
202
203 features::features_value_index_audit_range range = third.values_indices_audit();
204 features::iterator_all begin = range.begin();
205 if (same_namespace2) //next index differs for permutations and simple combinations
206 begin += (PROCESS_SELF_INTERACTIONS(ft_value)) ? j : j + 1;
207
208 features::iterator_all end = range.end();
209 inner_kernel<R, S, T, audit, audit_func>(dat, begin, end, offset, weights, ft_value, halfhash);
210 if (audit) audit_func(dat, nullptr);
211 } // end for (snd)
212 if(audit) audit_func(dat, nullptr);
213 } // end for (fst)
214
215 } // end if (data[thr] size > 0)
216 } // end if (data[snd] size > 0)
217 } // end if (data[fst] size > 0)
218
219 }
220 else // generic case: quatriples, etc.
221
222#endif
223 {
224
225 bool must_skip_interaction = false;
226 // preparing state data
227 feature_gen_data* fgd = state_data.begin();
228 feature_gen_data* fgd2; // for further use
229 for (namespace_index n : ns)
230 { features& ft = features_data[(int32_t)n];
231 const size_t ft_cnt = ft.indicies.size();
232
233 if (ft_cnt == 0)
234 { must_skip_interaction = true;
235 break;
236 }
237
238 if (fgd == state_data.end())
239 { state_data.push_back(empty_ns_data);
240 fgd = state_data.end()-1; // reassign as memory could be realloced
241 }
242
243 fgd->loop_end = ft_cnt-1; // saving number of features for each namespace
244 fgd->ft_arr = &ft;
245 ++fgd;
246 }
247
248 // if any of interacting namespace has 0 features - whole interaction is skipped
249 if (must_skip_interaction) continue; //no_data_to_interact
250
251 if (!all.permutations) // adjust state_data for simple combinations
252 { // if permutations mode is disabeled then namespaces in ns are already sorted and thus grouped
253 // (in fact, currently they are sorted even for enabled permutations mode)
254 // let's go throw the list and calculate number of features to skip in namespaces which
255 // repeated more than once to generate only simple combinations of features
256
257 size_t margin = 0; // number of features to ignore if namespace has been seen before
258
259 // iterate list backward as margin grows in this order
260
261 for (fgd = state_data.end()-1; fgd > state_data.begin(); --fgd)
262 { fgd2 = fgd-1;
263 fgd->self_interaction = (fgd->ft_arr == fgd2->ft_arr); //state_data.begin().self_interaction is always false
264 if (fgd->self_interaction)
265 { size_t& loop_end = fgd2->loop_end;
266
267 if (!PROCESS_SELF_INTERACTIONS((*fgd2->ft_arr).values[loop_end-margin]))
268 { ++margin; // otherwise margin can't be increased
269 if ( (must_skip_interaction = (loop_end < margin)) ) break;
270 }
271
272 if (margin != 0)
273 loop_end -= margin; // skip some features and increase margin
274 }
275 else if (margin != 0) margin = 0;
276 }
277
278 // if impossible_without_permutations == true then we faced with case like interaction 'aaaa'
279 // where namespace 'a' contains less than 4 unique features. It's impossible to make simple
280 // combination of length 4 without repetitions from 3 or less elements.
281 if (must_skip_interaction) continue; // impossible_without_permutations
282 } // end of state_data adjustment
283
284
285 fgd = state_data.begin(); // always equal to first ns
286 fgd2 = state_data.end()-1; // always equal to last ns
287 fgd->loop_idx = 0; // loop_idx contains current feature id for curently processed namespace.
288
289 // beware: micro-optimization.
290 /* start & end are always point to features in last namespace of interaction.
291 for 'all.permutations == true' they are constant.*/
292 size_t start_i = 0;
293
294 feature_gen_data* cur_data = fgd;
295 // end of micro-optimization block
296
297 // generic feature generation cycle for interactions of any length
298 bool do_it = true;
299 while (do_it)
300 { if (cur_data < fgd2) // can go further threw the list of namespaces in interaction
301 { feature_gen_data* next_data = cur_data+1;
302 size_t feature = cur_data->loop_idx;
303 features& fs = *(cur_data->ft_arr);
304
305 if (next_data->self_interaction)
306 { // if next namespace is same, we should start with loop_idx + 1 to avoid feature interaction with itself
307 // unless feature has value x and x != x*x. E.g. x != 0 and x != 1. Features with x == 0 are already
308 // filtered out in parce_args.cc::maybeFeature().
309
310 next_data->loop_idx = (PROCESS_SELF_INTERACTIONS(fs.values[feature])) ? cur_data->loop_idx : cur_data->loop_idx + 1;
311 }
312 else
313 next_data->loop_idx = 0;
314
315 if(audit) audit_func(dat, fs.space_names[feature].get());
316
317 if (cur_data == fgd) // first namespace
318 { next_data->hash = FNV_prime * (uint64_t)fs.indicies[feature];
319 next_data->x = fs.values[feature]; // data->x == 1.
320 }
321 else
322 { // feature2 xor (16777619*feature1)
323 next_data->hash = FNV_prime * (cur_data->hash ^ (uint64_t)fs.indicies[feature]);
324 next_data->x = INTERACTION_VALUE(fs.values[feature], cur_data->x);
325 }
326
327 ++cur_data;
328 }
329 else
330 { // last namespace - iterate its features and go back
331 if (!all.permutations) // start value is not a constant in this case
332 start_i = fgd2->loop_idx;
333
334 features& fs = *(fgd2->ft_arr);
335
336 feature_value ft_value = fgd2->x;
337 feature_index halfhash = fgd2->hash;
338
339 features::features_value_index_audit_range range = fs.values_indices_audit();
340 features::iterator_all begin = range.begin();
341 begin += start_i;
342 features::iterator_all end = range.begin();
343 end += fgd2->loop_end + 1;
344 inner_kernel<R, S, T, audit, audit_func, W>(dat, begin, end, offset, weights, ft_value, halfhash);
345
346 // trying to go back increasing loop_idx of each namespace by the way
347
348 bool go_further = true;
349
350 do
351 { --cur_data;
352 go_further = (++cur_data->loop_idx > cur_data->loop_end); //increment loop_idx
353 if (audit) audit_func(dat, nullptr);
354 }
355 while (go_further && cur_data != fgd);
356
357 do_it = !(cur_data == fgd && go_further);
358 //if do_it==false - we've reached 0 namespace but its 'cur_data.loop_idx > cur_data.loop_end' -> exit the while loop
359 } // if last namespace
360 } // while do_it
361 }
362 } // foreach interaction in all.interactions
363
364 state_data.delete_v();
365}

387inline long long choose(long long n, long long k)
388{ if (k > n) return 0;
389 if (k<0) return 0;
390 if (k==n) return 1;
391 if (k==0 && n!=0) return 1;
392 long long r = 1;
393 for (long long d = 1; d <= k; ++d)
394 { r *= n--;
395 r /= d;
396 }
397 return r;
398}
