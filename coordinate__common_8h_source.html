45inline double CoordinateDelta(double sum_grad, double sum_hess, double w,
46 double reg_alpha, double reg_lambda) {
47 if (sum_hess < 1e-5f) return 0.0f;
48 const double sum_grad_l2 = sum_grad + reg_lambda * w;
49 const double sum_hess_l2 = sum_hess + reg_lambda;
50 const double tmp = w - sum_grad_l2 / sum_hess_l2;
51 if (tmp >= 0) {
52 return std::max(-(sum_grad_l2 + reg_alpha) / sum_hess_l2, -w);
53 } else {
54 return std::min(-(sum_grad_l2 - reg_alpha) / sum_hess_l2, -w);
55 }
56}

66inline double CoordinateDeltaBias(double sum_grad, double sum_hess) {
67 return -sum_grad / sum_hess;
68}

81inline std::pair<double, double> GetGradient(Context const *ctx, int group_idx, int num_group,
82 bst_feature_t fidx,
83 std::vector<GradientPair> const &gpair,
84 DMatrix *p_fmat) {
85 double sum_grad = 0.0, sum_hess = 0.0;
86 for (const auto &batch : p_fmat->GetBatches<CSCPage>(ctx)) {
87 auto page = batch.GetView();
88 auto col = page[fidx];
89 const auto ndata = static_cast<bst_omp_uint>(col.size());
90 for (bst_omp_uint j = 0; j < ndata; ++j) {
91 const bst_float v = col[j].fvalue;
92 auto &p = gpair[col[j].index * num_group + group_idx];
93 if (p.GetHess() < 0.0f) continue;
94 sum_grad += p.GetGrad() * v;
95 sum_hess += p.GetHess() * v * v;
96 }
97 }
98 return std::make_pair(sum_grad, sum_hess);
99}

112inline std::pair<double, double> GetGradientParallel(Context const *ctx, int group_idx,
113 int num_group, int fidx,
114 const std::vector<GradientPair> &gpair,
115 DMatrix *p_fmat) {
116 std::vector<double> sum_grad_tloc(ctx->Threads(), 0.0);
117 std::vector<double> sum_hess_tloc(ctx->Threads(), 0.0);
118
119 for (const auto &batch : p_fmat->GetBatches<CSCPage>(ctx)) {
120 auto page = batch.GetView();
121 auto col = page[fidx];
122 const auto ndata = static_cast<bst_omp_uint>(col.size());
123 common::ParallelFor(ndata, ctx->Threads(), [&](size_t j) {
124 const bst_float v = col[j].fvalue;
125 auto &p = gpair[col[j].index * num_group + group_idx];
126 if (p.GetHess() < 0.0f) {
127 return;
128 }
129 auto t_idx = omp_get_thread_num();
130 sum_grad_tloc[t_idx] += p.GetGrad() * v;
131 sum_hess_tloc[t_idx] += p.GetHess() * v * v;
132 });
133 }
134 double sum_grad =
135 std::accumulate(sum_grad_tloc.cbegin(), sum_grad_tloc.cend(), 0.0);
136 double sum_hess =
137 std::accumulate(sum_hess_tloc.cbegin(), sum_hess_tloc.cend(), 0.0);
138 return std::make_pair(sum_grad, sum_hess);
139}

151inline std::pair<double, double> GetBiasGradientParallel(int group_idx, int num_group,
152 const std::vector<GradientPair> &gpair,
153 DMatrix *p_fmat, int32_t n_threads) {
154 const auto ndata = static_cast<bst_omp_uint>(p_fmat->Info().num_row_);
155 std::vector<double> sum_grad_tloc(n_threads, 0);
156 std::vector<double> sum_hess_tloc(n_threads, 0);
157
158 common::ParallelFor(ndata, n_threads, [&](auto i) {
159 auto tid = omp_get_thread_num();
160 auto &p = gpair[i * num_group + group_idx];
161 if (p.GetHess() >= 0.0f) {
162 sum_grad_tloc[tid] += p.GetGrad();
163 sum_hess_tloc[tid] += p.GetHess();
164 }
165 });
166 double sum_grad = std::accumulate(sum_grad_tloc.cbegin(), sum_grad_tloc.cend(), 0.0);
167 double sum_hess = std::accumulate(sum_hess_tloc.cbegin(), sum_hess_tloc.cend(), 0.0);
168 return std::make_pair(sum_grad, sum_hess);
169}

181inline void UpdateResidualParallel(Context const *ctx, bst_feature_t fidx, int group_idx,
182 int num_group, float dw, std::vector<GradientPair> *in_gpair,
183 DMatrix *p_fmat) {
184 if (dw == 0.0f) return;
185 for (const auto &batch : p_fmat->GetBatches<CSCPage>(ctx)) {
186 auto page = batch.GetView();
187 auto col = page[fidx];
188 // update grad value
189 const auto num_row = static_cast<bst_omp_uint>(col.size());
190 common::ParallelFor(num_row, ctx->Threads(), [&](auto j) {
191 GradientPair &p = (*in_gpair)[col[j].index * num_group + group_idx];
192 if (p.GetHess() < 0.0f) return;
193 p += GradientPair(p.GetHess() * col[j].fvalue * dw, 0);
194 });
195 }
196}

208 float dbias, std::vector<GradientPair> *in_gpair,

214 if (g.GetHess() < 0.0f) return;
215 g += GradientPair(g.GetHess() * dbias, 0);
216 });

321class GreedyFeatureSelector : public FeatureSelector {
322 public:
323 using FeatureSelector::FeatureSelector;
324 void Setup(Context const *, const gbm::GBLinearModel &model, const std::vector<GradientPair> &,
325 DMatrix *, float, float, int param) override {
326 top_k_ = static_cast<bst_uint>(param);
327 const bst_uint ngroup = model.learner_model_param->num_output_group;
328 if (param <= 0) top_k_ = std::numeric_limits<bst_uint>::max();
329 if (counter_.size() == 0) {
330 counter_.resize(ngroup);
331 gpair_sums_.resize(model.learner_model_param->num_feature * ngroup);
332 }
333 for (bst_uint gid = 0u; gid < ngroup; ++gid) {
334 counter_[gid] = 0u;
335 }
336 }
337
338 int NextFeature(Context const* ctx, int, const gbm::GBLinearModel &model,
339 int group_idx, const std::vector<GradientPair> &gpair,
340 DMatrix *p_fmat, float alpha, float lambda) override {
341 // k-th selected feature for a group
342 auto k = counter_[group_idx]++;
343 // stop after either reaching top-K or going through all the features in a group
344 if (k >= top_k_ || counter_[group_idx] == model.learner_model_param->num_feature) return -1;
345
346 const int ngroup = model.learner_model_param->num_output_group;
347 const bst_omp_uint nfeat = model.learner_model_param->num_feature;
348 // Calculate univariate gradient sums
349 std::fill(gpair_sums_.begin(), gpair_sums_.end(), std::make_pair(0., 0.));
350 for (const auto &batch : p_fmat->GetBatches<CSCPage>(ctx)) {
351 auto page = batch.GetView();
352 common::ParallelFor(nfeat, ctx->Threads(), [&](bst_omp_uint i) {
353 const auto col = page[i];
354 const bst_uint ndata = col.size();
355 auto &sums = gpair_sums_[group_idx * nfeat + i];
356 for (bst_uint j = 0u; j < ndata; ++j) {
357 const bst_float v = col[j].fvalue;
358 auto &p = gpair[col[j].index * ngroup + group_idx];
359 if (p.GetHess() < 0.f) continue;
360 sums.first += p.GetGrad() * v;
361 sums.second += p.GetHess() * v * v;
362 }
363 });
364 }
365 // Find a feature with the largest magnitude of weight change
366 int best_fidx = 0;
367 double best_weight_update = 0.0f;
368 for (bst_omp_uint fidx = 0; fidx < nfeat; ++fidx) {
369 auto &s = gpair_sums_[group_idx * nfeat + fidx];
370 float dw = std::abs(static_cast<bst_float>(
371 CoordinateDelta(s.first, s.second, model[fidx][group_idx], alpha, lambda)));
372 if (dw > best_weight_update) {
373 best_weight_update = dw;
374 best_fidx = fidx;
375 }
376 }
377 return best_fidx;
378 }
379
380 protected:
381 bst_uint top_k_;
382 std::vector<bst_uint> counter_;
383 std::vector<std::pair<double, double>> gpair_sums_;
384};

397class ThriftyFeatureSelector : public FeatureSelector {
398 public:
399 using FeatureSelector::FeatureSelector;
400
401 void Setup(Context const *ctx, const gbm::GBLinearModel &model,
402 const std::vector<GradientPair> &gpair, DMatrix *p_fmat, float alpha, float lambda,
403 int param) override {
404 top_k_ = static_cast<bst_uint>(param);
405 if (param <= 0) top_k_ = std::numeric_limits<bst_uint>::max();
406 const bst_uint ngroup = model.learner_model_param->num_output_group;
407 const bst_omp_uint nfeat = model.learner_model_param->num_feature;
408
409 if (deltaw_.size() == 0) {
410 deltaw_.resize(nfeat * ngroup);
411 sorted_idx_.resize(nfeat * ngroup);
412 counter_.resize(ngroup);
413 gpair_sums_.resize(nfeat * ngroup);
414 }
415 // Calculate univariate gradient sums
416 std::fill(gpair_sums_.begin(), gpair_sums_.end(), std::make_pair(0., 0.));
417 for (const auto &batch : p_fmat->GetBatches<CSCPage>(ctx)) {
418 auto page = batch.GetView();
419 // column-parallel is usually fastaer than row-parallel
420 common::ParallelFor(nfeat, ctx->Threads(), [&](auto i) {
421 const auto col = page[i];
422 const bst_uint ndata = col.size();
423 for (bst_uint gid = 0u; gid < ngroup; ++gid) {
424 auto &sums = gpair_sums_[gid * nfeat + i];
425 for (bst_uint j = 0u; j < ndata; ++j) {
426 const bst_float v = col[j].fvalue;
427 auto &p = gpair[col[j].index * ngroup + gid];
428 if (p.GetHess() < 0.f) continue;
429 sums.first += p.GetGrad() * v;
430 sums.second += p.GetHess() * v * v;
431 }
432 }
433 });
434 }
435 // rank by descending weight magnitude within the groups
436 std::fill(deltaw_.begin(), deltaw_.end(), 0.f);
437 std::iota(sorted_idx_.begin(), sorted_idx_.end(), 0);
438 bst_float *pdeltaw = &deltaw_[0];
439 for (bst_uint gid = 0u; gid < ngroup; ++gid) {
440 // Calculate univariate weight changes
441 for (bst_omp_uint i = 0; i < nfeat; ++i) {
442 auto ii = gid * nfeat + i;
443 auto &s = gpair_sums_[ii];
444 deltaw_[ii] = static_cast<bst_float>(CoordinateDelta(
445 s.first, s.second, model[i][gid], alpha, lambda));
446 }
447 // sort in descending order of deltaw abs values
448 auto start = sorted_idx_.begin() + gid * nfeat;
449 std::sort(start, start + nfeat,
450 [pdeltaw](size_t i, size_t j) {
451 return std::abs(*(pdeltaw + i)) > std::abs(*(pdeltaw + j));
452 });
453 counter_[gid] = 0u;
454 }
455 }
456
457 int NextFeature(Context const *, int, const gbm::GBLinearModel &model, int group_idx,
458 const std::vector<GradientPair> &, DMatrix *, float, float) override {
459 // k-th selected feature for a group
460 auto k = counter_[group_idx]++;
461 // stop after either reaching top-N or going through all the features in a group
462 if (k >= top_k_ || counter_[group_idx] == model.learner_model_param->num_feature) return -1;
463 // note that sorted_idx stores the "long" indices
464 const size_t grp_offset = group_idx * model.learner_model_param->num_feature;
465 return static_cast<int>(sorted_idx_[grp_offset + k] - grp_offset);
466 }
467
468 protected:
469 bst_uint top_k_;
470 std::vector<bst_float> deltaw_;
471 std::vector<size_t> sorted_idx_;
472 std::vector<bst_uint> counter_;
473 std::vector<std::pair<double, double>> gpair_sums_;
474};

476inline FeatureSelector *FeatureSelector::Create(int choice) {
477 switch (choice) {
478 case kCyclic:
479 return new CyclicFeatureSelector;
480 case kShuffle:
481 return new ShuffleFeatureSelector;

485 return new GreedyFeatureSelector;
486 case kRandom:
487 return new RandomFeatureSelector;
488 default:
489 LOG(FATAL) << "unknown coordinate selector: " << choice;
490 }
491 return nullptr;
