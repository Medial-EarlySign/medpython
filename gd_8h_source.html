28 struct multipredict_info { size_t count; size_t step; polyprediction* pred; const T& weights; /* & for l1: */ float gravity; };

30template<class T>
31inline void vec_add_multipredict(multipredict_info<T>& mp, const float fx, uint64_t fi)
32{
33 if ((-1e-10 < fx) && (fx < 1e-10)) return;
34 uint64_t mask = mp.weights.mask();
35 polyprediction* p = mp.pred;
36 fi &= mask;
37 uint64_t top = fi + (uint64_t)((mp.count - 1) * mp.step);
38 uint64_t i = 0;
39 if (top <= mask)
40 {
41 i += fi;
42 for (; i <= top; i += mp.step, ++p)
43 p->scalar += fx * mp.weights[i]; //TODO: figure out how to use weight_parameters::iterator (not using change_begin())
44 }
45 else // TODO: this could be faster by unrolling into two loops
46 for (size_t c = 0; c<mp.count; ++c, fi += (uint64_t)mp.step, ++p)
47 {
48 fi &= mask;
49 p->scalar += fx * mp.weights[fi];
50 }
51}

90// iterate through all namespaces and quadratic&cubic features, callback function T(some_data_R, feature_value_x, S)
91// where S is EITHER float& feature_weight OR uint64_t feature_index
92template <class R, class S, void (*T)(R&, float, S)>
93inline void foreach_feature(vw& all, example& ec, R& dat)
94{ uint64_t offset = ec.ft_offset;
95 if (all.weights.sparse)
96 if (all.ignore_some_linear)
97 for (example::iterator i = ec.begin (); i != ec.end(); ++i)
98 {
99 if (!all.ignore_linear[i.index()])
100 {
101 features& f = *i;
102 foreach_feature<R, T, sparse_parameters>(all.weights.sparse_weights, f, dat, offset);
103 }
104 }
105 else
106 for (features& f : ec)
107 foreach_feature<R, T, sparse_parameters>(all.weights.sparse_weights, f, dat, offset);
108 else
109 if (all.ignore_some_linear)
110 for (example::iterator i = ec.begin (); i != ec.end(); ++i)
111 {
112 if (!all.ignore_linear[i.index()])
113 {
114 features& f = *i;
115 foreach_feature<R, T, dense_parameters>(all.weights.dense_weights, f, dat, offset);
116 }
117 }
118 else
119 for (features& f : ec)
120 foreach_feature<R, T, dense_parameters>(all.weights.dense_weights, f, dat, offset);
121
122 INTERACTIONS::generate_interactions<R,S,T>(all, ec, dat);
123}
