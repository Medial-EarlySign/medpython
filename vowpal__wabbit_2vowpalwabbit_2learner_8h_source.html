100struct learner
101{
102private:
103 func_data init_fd;
104 learn_data learn_fd;
105 sensitivity_data sensitivity_fd;
106 finish_example_data finish_example_fd;
107 save_load_data save_load_fd;
108 func_data end_pass_fd;
109 func_data end_examples_fd;
110 func_data finisher_fd;
111
112public:
113 prediction_type::prediction_type_t pred_type;
114 size_t weights; //this stores the number of "weight vectors" required by the learner.
115 size_t increment;
116
117 //called once for each example. Must work under reduction.
118 inline void learn(example& ec, size_t i=0)
119 { ec.ft_offset += (uint32_t)(increment*i);
120 learn_fd.learn_f(learn_fd.data, *learn_fd.base, ec);
121 ec.ft_offset -= (uint32_t)(increment*i);
122 }
123 inline void predict(example& ec, size_t i=0)
124 { ec.ft_offset += (uint32_t)(increment*i);
125 learn_fd.predict_f(learn_fd.data, *learn_fd.base, ec);
126 ec.ft_offset -= (uint32_t)(increment*i);
127 }
128 inline void multipredict(example& ec, size_t lo, size_t count, polyprediction* pred, bool finalize_predictions)
129 { if (learn_fd.multipredict_f == NULL)
130 { ec.ft_offset += (uint32_t)(increment*lo);
131 for (size_t c=0; c<count; c++)
132 { learn_fd.predict_f(learn_fd.data, *learn_fd.base, ec);
133 if (finalize_predictions) pred[c] = ec.pred; // TODO: this breaks for complex labels because = doesn't do deep copy!
134 else pred[c].scalar = ec.partial_prediction;
135 //pred[c].scalar = finalize_prediction ec.partial_prediction; // TODO: this breaks for complex labels because = doesn't do deep copy! // note works if ec.partial_prediction, but only if finalize_prediction is run????
136 ec.ft_offset += (uint32_t)increment;
137 }
138 ec.ft_offset -= (uint32_t)(increment*(lo+count));
139 }
140 else
141 { ec.ft_offset += (uint32_t)(increment*lo);
142 learn_fd.multipredict_f(learn_fd.data, *learn_fd.base, ec, count, increment, pred, finalize_predictions);
143 ec.ft_offset -= (uint32_t)(increment*lo);
144 }
145 }
146 inline void set_predict(void (*u)(T& data, base_learner& base, example&)) { learn_fd.predict_f = (tlearn)u; }
147 inline void set_learn(void (*u)(T&, base_learner&, example&)) { learn_fd.learn_f = (tlearn)u; }
148 inline void set_multipredict(void (*u)(T&, base_learner&, example&, size_t, size_t, polyprediction*, bool)) { learn_fd.multipredict_f = (tmultipredict)u; }
149
150 inline void update(example& ec, size_t i=0)
151 { ec.ft_offset += (uint32_t)(increment*i);
152 learn_fd.update_f(learn_fd.data, *learn_fd.base, ec);
153 ec.ft_offset -= (uint32_t)(increment*i);
154 }
155 inline void set_update(void (*u)(T& data, base_learner& base, example&))
156 { learn_fd.update_f = (tlearn)u; }
157
158 //used for active learning and confidence to determine how easily predictions are changed
159 inline void set_sensitivity(float (*u)(T& data, base_learner& base, example&))
160 { sensitivity_fd.data = learn_fd.data;
161 sensitivity_fd.sensitivity_f = (tsensitivity)u;
162 }
163 inline float sensitivity(example& ec, size_t i=0)
164 { ec.ft_offset += (uint32_t)(increment*i);
165 float ret = sensitivity_fd.sensitivity_f(sensitivity_fd.data, *learn_fd.base, ec);
166 ec.ft_offset -= (uint32_t)(increment*i);
167 return ret;
168 }
169
170 //called anytime saving or loading needs to happen. Autorecursive.
171 inline void save_load(io_buf& io, bool read, bool text)
172 { save_load_fd.save_load_f(save_load_fd.data, io, read, text);
173 if (save_load_fd.base) save_load_fd.base->save_load(io, read, text);
174 }
175 inline void set_save_load(void (*sl)(T&, io_buf&, bool, bool))
176 { save_load_fd.save_load_f = (tsl)sl;
177 save_load_fd.data = learn_fd.data;
178 save_load_fd.base = learn_fd.base;
179 }
180
181 //called to clean up state. Autorecursive.
182 void set_finish(void (*f)(T&))
183 { finisher_fd = tuple_dbf(learn_fd.data,learn_fd.base, (tfunc)f); }
184 inline void finish()
185 { if (finisher_fd.data)
186 {finisher_fd.func(finisher_fd.data); free(finisher_fd.data); }
187 if (finisher_fd.base)
188 { finisher_fd.base->finish();
189 free(finisher_fd.base);
190 }
191 }
192
193 void end_pass()
194 { end_pass_fd.func(end_pass_fd.data);
195 if (end_pass_fd.base) end_pass_fd.base->end_pass();
196 }//autorecursive
197 void set_end_pass(void (*f)(T&))
198 {end_pass_fd = tuple_dbf(learn_fd.data, learn_fd.base, (tfunc)f);}
199
200 //called after parsing of examples is complete. Autorecursive.
201 void end_examples()
202 { end_examples_fd.func(end_examples_fd.data);
203 if (end_examples_fd.base) end_examples_fd.base->end_examples();
204 }
205 void set_end_examples(void (*f)(T&))
206 {end_examples_fd = tuple_dbf(learn_fd.data,learn_fd.base, (tfunc)f);}

208 //Called at the beginning by the driver. Explicitly not recursive.
209 void init_driver() { init_fd.func(init_fd.data);}
210 void set_init_driver(void (*f)(T&))
211 { init_fd = tuple_dbf(learn_fd.data,learn_fd.base, (tfunc)f); }

213 //called after learn example for each example. Explicitly not recursive.
214 inline void finish_example(vw& all, example& ec)
215 { finish_example_fd.finish_example_f(all, finish_example_fd.data, ec);}
216 void set_finish_example(void (*f)(vw& all, T&, example&))
217 { finish_example_fd.data = learn_fd.data;

229template<class T>
230learner<T>& init_learner(T* dat, void (*learn)(T&, base_learner&, example&),
231 size_t params_per_weight, prediction_type::prediction_type_t pred_type)
232{ // the constructor for all learning algorithms.
233 learner<T>& ret = calloc_or_throw<learner<T> >();
234 ret.weights = 1;
235 ret.increment = params_per_weight;
236 ret.end_pass_fd.func = noop;
237 ret.end_examples_fd.func = noop;
238 ret.init_fd.func = noop;
239 ret.save_load_fd.save_load_f = noop_sl;
240 ret.finisher_fd.data = dat;
241 ret.finisher_fd.func = noop;
242
243 ret.learn_fd.data = dat;
244 ret.learn_fd.learn_f = (tlearn)learn;
245 ret.learn_fd.update_f = (tlearn)learn;
246 ret.learn_fd.predict_f = (tlearn)learn;
247 ret.learn_fd.multipredict_f = nullptr;
248 ret.sensitivity_fd.sensitivity_f = (tsensitivity)noop_sensitivity;
249 ret.finish_example_fd.data = dat;
250 ret.finish_example_fd.finish_example_f = return_simple_example;
251 ret.pred_type = pred_type;
252
253 return ret;
254}

263template<class T>
264learner<T>& init_learner(T* dat, base_learner* base,
265 void (*learn)(T&, base_learner&, example&),
266 void (*predict)(T&, base_learner&, example&), size_t ws,
267 prediction_type::prediction_type_t pred_type)
268{ //the reduction constructor, with separate learn and predict functions
269 learner<T>& ret = calloc_or_throw<learner<T> >();
270 ret = *(learner<T>*)base;
271
272 ret.learn_fd.data = dat;
273 ret.learn_fd.learn_f = (tlearn)learn;
274 ret.learn_fd.update_f = (tlearn)learn;
275 ret.learn_fd.predict_f = (tlearn)predict;
276 ret.learn_fd.multipredict_f = nullptr;
277 ret.learn_fd.base = base;
278
279 ret.finisher_fd.data = dat;
280 ret.finisher_fd.base = base;
281 ret.finisher_fd.func = noop;
282 ret.pred_type = pred_type;
283
284 ret.weights = ws;
285 ret.increment = base->increment * ret.weights;
286 return ret;
287}
