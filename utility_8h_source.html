89struct murmur_hash
90{
91
92 //-----------------------------------------------------------------------------
93 // Block read - if your platform needs to do endian-swapping or can only
94 // handle aligned reads, do the conversion here
95private:
96 static inline uint32_t getblock(const uint32_t * p, int i)
97 { return p[i];
98 }
99
100 //-----------------------------------------------------------------------------
101 // Finalization mix - force all bits of a hash block to avalanche
102
103 static inline uint32_t fmix(uint32_t h)
104 { h ^= h >> 16;
105 h *= 0x85ebca6b;
106 h ^= h >> 13;
107 h *= 0xc2b2ae35;
108 h ^= h >> 16;
109
110 return h;
111 }
112
113 //-----------------------------------------------------------------------------
114public:
115 uint32_t uniform_hash(const void * key, size_t len, uint32_t seed)
116 { const uint8_t * data = (const uint8_t*)key;
117 const int nblocks = (int)len / 4;
118
119 uint32_t h1 = seed;
120
121 const uint32_t c1 = 0xcc9e2d51;
122 const uint32_t c2 = 0x1b873593;
123
124 // --- body
125 const uint32_t * blocks = (const uint32_t *)(data + nblocks * 4);
126
127 for (int i = -nblocks; i; i++)
128 { uint32_t k1 = getblock(blocks, i);
129
130 k1 *= c1;
131 k1 = ROTL32(k1, 15);
132 k1 *= c2;
133
134 h1 ^= k1;
135 h1 = ROTL32(h1, 13);
136 h1 = h1 * 5 + 0xe6546b64;
137 }
138
139 // --- tail
140 const uint8_t * tail = (const uint8_t*)(data + nblocks * 4);
141
142 uint32_t k1 = 0;
143
144 switch (len & 3)
145 { case 3: k1 ^= tail[2] << 16;
146 case 2: k1 ^= tail[1] << 8;
147 case 1: k1 ^= tail[0];
148 k1 *= c1; k1 = ROTL32(k1, 15); k1 *= c2; h1 ^= k1;
149 }
150
151 // --- finalization
152 h1 ^= len;
153
154 return fmix(h1);
155 }
156};

175const size_t max_int = 100000;
176const float max_float = max_int;
177const float min_float = 0.00001f;
178const size_t max_digits = (size_t) roundf((float) (-log(min_float) / log(10.)));
179
180class NumberUtils
181{
182public:
183 template<bool trailing_zeros>
184 static void print_mantissa(char*& begin, float f)
185 { // helper for print_float
186 char values[10];
187 size_t v = (size_t)f;
188 size_t digit = 0;
189 size_t first_nonzero = 0;
190 for (size_t max = 1; max <= v; ++digit)
191 { size_t max_next = max * 10;
192 char v_mod = (char) (v % max_next / max);
193 if (!trailing_zeros && v_mod != '\0' && first_nonzero == 0)
194 first_nonzero = digit;
195 values[digit] = '0' + v_mod;
196 max = max_next;
197 }
198 if (!trailing_zeros)
199 for (size_t i = max_digits; i > digit; i--)
200 *begin++ = '0';
201 while (digit > first_nonzero)
202 *begin++ = values[--digit];
203 }
204
205 static void print_float(char* begin, size_t size, float f)
206 { bool sign = false;
207 if (f < 0.f)
208 sign = true;
209 float unsigned_f = fabsf(f);
210 if (unsigned_f < max_float && unsigned_f > min_float)
211 { if (sign)
212 *begin++ = '-';
213 print_mantissa<true>(begin, unsigned_f);
214 unsigned_f -= (size_t)unsigned_f;
215 unsigned_f *= max_int;
216 if (unsigned_f >= 1.f)
217 { *begin++ = '.';
218 print_mantissa<false>(begin, unsigned_f);
219 }
220 }
221 else if (unsigned_f == 0.)
222 *begin++ = '0';
223 else
224 { sprintf_s(begin, size, "%g", f);
225 return;
226 }
227 *begin = '\0';
228 return;
229 }
230};

243//A quick implementation similar to drand48 for cross-platform compatibility
244namespace PRG
245{
246const uint64_t a = 0xeece66d5deece66dULL;
247const uint64_t c = 2147483647;
248
249const int bias = 127 << 23;
250
251union int_float
252{ int32_t i;
253 float f;
254};
255
256struct prg
257{
258private:
259 uint64_t v;
260public:
261 prg() { v = c; }
262 prg(uint64_t initial) { v = initial; }
263
264 float merand48(uint64_t& initial)
265 { initial = a * initial + c;
266 int_float temp;
267 temp.i = ((initial >> 25) & 0x7FFFFF) | bias;
268 return temp.f - 1;
269 }
270
271 float Uniform_Unit_Interval()
272 { return merand48(v);
273 }
274
275 uint32_t Uniform_Int(uint32_t low, uint32_t high)
276 { merand48(v);
277 uint32_t ret = low + ((v >> 25) % (high - low + 1));
278 return ret;
279 }
280};
